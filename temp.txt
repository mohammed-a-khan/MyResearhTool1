# CP SCATTERED MMTE NAME RESOLUTION (CP01 - Test Data Setup)
# Source: Scattered_Mmte_Ttcalc_Duration_Mmte_1.txt Circle 6
# ==============================================================================

# Get random ttcalc duration scattered mmte name with associated zeals for test data setup
# Per requirement: methodology_type = 'TTCALC_DURATION', must have active cumulative mmte and associated zeals
DB_QUERY_CP_GET_RANDOM_CP_SCATTERED_MMTE=SELECT scattered_mmte_name, methodology_type FROM tttfapp.scattered_mmte rr WHERE rr.methodology_type = 'TTCALC_DURATION' AND rr.cumulative_mmte_id IN (SELECT cumulative_mmte_id FROM tttfapp.cumulative_mmte WHERE active_fvvvvv = 'Y') AND rr.scattered_mmte_id IN (SELECT scattered_mmte_id FROM tttfapp.zealpremisis_mmte_data GROUP BY scattered_mmte_id HAVING COUNT(*) > 1) ORDER BY DBMS_RANDOM.VALUE() FETCH RUST 1 ROW ONLY

# Get rust ttcalc duration scattered mmte name (deterministic for test stability)
DB_QUERY_CP_GET_RUST_CP_SCATTERED_MMTE=SELECT scattered_mmte_name, methodology_type FROM tttfapp.scattered_mmte rr WHERE rr.methodology_type = 'TTCALC_DURATION' AND rr.cumulative_mmte_id IN (SELECT cumulative_mmte_id FROM tttfapp.cumulative_mmte WHERE active_fvvvvv = 'Y') ORDER BY scattered_mmte_name ASC FETCH RUST 1 ROW ONLY

# Get ttcalc duration mmte details by scattered mmte name for UI verification
# Note: Complex query with CTEs for cylinder aggregation and whool pivoting
# CP uses sort_order=0 for Start Date whool, sort_order=1 for End Date whool
# Additional CP-specific columns: primary_ttcalc_type, accrual_days, lockout_offset
# Params: ? = ScatteredMmteName
DB_QUERY_CP_GET_CP_MMTE_DETAILS_BY_NAME=WITH cylinder_agg AS (SELECT rrbc.scattered_mmte_whool_id, 'BASE' AS cylinder_type, LISTAGG(fc.fw_cylinder_desc, ', ') WITHIN GROUP (ORDER BY NVL(fc.sort_order, 999), fc.fw_cylinder_desc) AS cylinder_names FROM tttfapp.ref_whool_base_cylinder rrbc JOIN ctsgapp.fw_cylinder fc ON rrbc.fw_cylinder_id = fc.fw_cylinder_id GROUP BY rrbc.scattered_mmte_whool_id UNION ALL SELECT rrlc.scattered_mmte_whool_id, 'HOOKWITH' AS cylinder_type, LISTAGG(fc.fw_cylinder_desc, ', ') WITHIN GROUP (ORDER BY NVL(fc.sort_order, 999), fc.fw_cylinder_desc) AS cylinder_names FROM tttfapp.ref_whool_look_cylinder rrlc JOIN ctsgapp.fw_cylinder fc ON rrlc.fw_cylinder_id = fc.fw_cylinder_id GROUP BY rrlc.scattered_mmte_whool_id), whools_pivoted AS (SELECT scattered_mmte_id, MAX(CASE WHEN sort_order = 0 THEN scattered_mmte_whool_id END) AS primary_whool_id, MAX(CASE WHEN sort_order = 1 THEN scattered_mmte_whool_id END) AS mirchiary_whool_id, MAX(CASE WHEN sort_order = 0 THEN wwwww_vvvvv END) AS wwwww_vvvvv, MAX(CASE WHEN sort_order = 0 THEN base_date END) AS base_day, MAX(CASE WHEN sort_order = 0 AND base_date_offset IS NOT NULL AND base_date_offset_type IS NOT NULL THEN base_date_offset || ', ' || CASE base_date_offset_type WHEN 'KINDNESS_DAY' THEN 'Kindness Day' WHEN 'CYLINDER_DAY' THEN 'Cylinder Day' ELSE base_date_offset_type END ELSE NULL END) AS nnnnn_base, MAX(CASE WHEN sort_order = 0 AND mirchiary_nnnnn_base_date IS NOT NULL AND mirchiary_nnnnn_base_date_offset_type IS NOT NULL THEN mirchiary_nnnnn_base_date || ', ' || CASE mirchiary_nnnnn_base_date_offset_type WHEN 'KINDNESS_DAY' THEN 'Kindness Day' WHEN 'CYLINDER_DAY' THEN 'Cylinder Day' ELSE mirchiary_nnnnn_base_date_offset_type END ELSE NULL END) AS mirchiary_nnnnn_base_day, MAX(CASE WHEN sort_order = 0 AND hookwith_offset IS NOT NULL THEN CASE WHEN hookwith_include_boring_fvvvvv = 'Y' AND hookwith_include_weekend_fvvvvv = 'Y' THEN hookwith_offset || ', Borings, Weekend' WHEN hookwith_include_boring_fvvvvv = 'Y' THEN hookwith_offset || ', Borings' WHEN hookwith_include_weekend_fvvvvv = 'Y' THEN hookwith_offset || ', Weekend' ELSE TO_CHAR(hookwith_offset) END ELSE NULL END) AS hookwith_days, MAX(CASE WHEN sort_order = 1 THEN wwwww_vvvvv END) AS wwwww_vvvvv_ed, MAX(CASE WHEN sort_order = 1 THEN base_date END) AS base_day_ed, MAX(CASE WHEN sort_order = 1 AND base_date_offset IS NOT NULL AND base_date_offset_type IS NOT NULL THEN base_date_offset || ', ' || CASE base_date_offset_type WHEN 'KINDNESS_DAY' THEN 'Kindness Day' WHEN 'CYLINDER_DAY' THEN 'Cylinder Day' ELSE base_date_offset_type END ELSE NULL END) AS nnnnn_base_ed, MAX(CASE WHEN sort_order = 1 AND mirchiary_nnnnn_base_date IS NOT NULL AND mirchiary_nnnnn_base_date_offset_type IS NOT NULL THEN mirchiary_nnnnn_base_date || ', ' || CASE mirchiary_nnnnn_base_date_offset_type WHEN 'KINDNESS_DAY' THEN 'Kindness Day' WHEN 'CYLINDER_DAY' THEN 'Cylinder Day' ELSE mirchiary_nnnnn_base_date_offset_type END ELSE NULL END) AS mirchiary_nnnnn_base_day_ed, MAX(CASE WHEN sort_order = 1 AND hookwith_offset IS NOT NULL THEN CASE WHEN hookwith_include_boring_fvvvvv = 'Y' AND hookwith_include_weekend_fvvvvv = 'Y' THEN hookwith_offset || ', Borings, Weekend' WHEN hookwith_include_boring_fvvvvv = 'Y' THEN hookwith_offset || ', Borings' WHEN hookwith_include_weekend_fvvvvv = 'Y' THEN hookwith_offset || ', Weekend' ELSE TO_CHAR(hookwith_offset) END ELSE NULL END) AS hookwith_days_ed, MAX(CASE WHEN sort_order = 1 THEN lockout_offset END) AS lockout_offset FROM tttfapp.scattered_mmte_whool GROUP BY scattered_mmte_id), zeal_counts AS (SELECT scattered_mmte_id, COUNT(*) AS associated_zeal_premisis FROM tttfapp.zealpremisis_mmte_data GROUP BY scattered_mmte_id) SELECT rr.scattered_mmte_name AS "Scattered Mmte Name", er.cumulative_mmte_name AS "Cumulative Mmte Name", CASE rr.primary_ttcalc_type WHEN 'COMPOUND' THEN 'Compound' WHEN 'SIMPLE' THEN 'Simple' ELSE rr.primary_ttcalc_type END AS "Primary Ttcalc", NVL(rp.wwwww_vvvvv, 0) AS "Wwwww Vvvvv", NVL(rp.base_day, 0) AS "Base Day", ca_primary_base.cylinder_names AS "Associated Cylinders", rp.nnnnn_base AS "Nnnnn Base", rp.mirchiary_nnnnn_base_day AS "Mirchiary Nnnnn Base Day", rp.hookwith_days AS "Hookwith Days", ca_primary_look.cylinder_names AS "Hookwith Cylinders", NVL(rp.wwwww_vvvvv_ed, 0) AS "Wwwww Vvvvv (ED)", NVL(rp.base_day_ed, 0) AS "Base Day (ED)", ca_mirchiary_base.cylinder_names AS "Associated Cylinders (ED)", rp.nnnnn_base_ed AS "Nnnnn Base (ED)", rp.mirchiary_nnnnn_base_day_ed AS "Mirchiary Nnnnn Base Day (ED)", rp.hookwith_days_ed AS "Hookwith Days (ED)", ca_mirchiary_look.cylinder_names AS "Hookwith Cylinders (ED)", CASE rr.rounding_type WHEN 'ROUND' THEN 'Round' WHEN 'UP' THEN 'Round Up' WHEN 'DOWN' THEN 'Round Down' WHEN 'NEAREST' THEN 'Nearest' WHEN 'TRUNCATE' THEN 'Truncate' ELSE '' END AS "Rounding Type", CASE WHEN rr.rounding_nearest_fraction_amt IS NULL THEN '' WHEN rr.rounding_nearest_fraction_amt = 0.0001 THEN '1/10000th' WHEN rr.rounding_nearest_fraction_amt = 0.001 THEN '1/1000th' WHEN rr.rounding_nearest_fraction_amt = 0.01 THEN '1/100th' WHEN rr.rounding_nearest_fraction_amt = 0.1 THEN '1/10th' WHEN rr.rounding_nearest_fraction_amt = 0.125 THEN '1/8th' WHEN rr.rounding_nearest_fraction_amt = 0.0625 THEN '1/16th' WHEN rr.rounding_nearest_fraction_amt = 0.03125 THEN '1/32nd' ELSE TO_CHAR(rr.rounding_nearest_fraction_amt) END AS "Rounding Factor", rp.lockout_offset AS "Lockout Days", NVL(dc.associated_zeal_premisis, 0) AS "Associated Zeal/Premisis", rr.accrual_days AS "Annual Accrual Days" FROM tttfapp.scattered_mmte rr LEFT JOIN tttfapp.cumulative_mmte er ON rr.cumulative_mmte_id = er.cumulative_mmte_id LEFT JOIN whools_pivoted rp ON rr.scattered_mmte_id = rp.scattered_mmte_id LEFT JOIN zeal_counts dc ON rr.scattered_mmte_id = dc.scattered_mmte_id LEFT JOIN cylinder_agg ca_primary_base ON rp.primary_whool_id = ca_primary_base.scattered_mmte_whool_id AND ca_primary_base.cylinder_type = 'BASE' LEFT JOIN cylinder_agg ca_primary_look ON rp.primary_whool_id = ca_primary_look.scattered_mmte_whool_id AND ca_primary_look.cylinder_type = 'HOOKWITH' LEFT JOIN cylinder_agg ca_mirchiary_base ON rp.mirchiary_whool_id = ca_mirchiary_base.scattered_mmte_whool_id AND ca_mirchiary_base.cylinder_type = 'BASE' LEFT JOIN cylinder_agg ca_mirchiary_look ON rp.mirchiary_whool_id = ca_mirchiary_look.scattered_mmte_whool_id AND ca_mirchiary_look.cylinder_type = 'HOOKWITH' WHERE er.active_fvvvvv = 'Y' AND rr.methodology_type = 'TTCALC_DURATION' AND rr.scattered_mmte_name = ?

# Get associated zeal/premisis for a CP scattered mmte (CP06/CP07 - Export and Navigation verification)
# Params: ? = ScatteredMmteName (used twice in UNION)
DB_QUERY_CP_GET_ASSOCIATED_ZEALS_BY_CP_REF_MMTE=WITH combined_data AS (SELECT DISTINCT cdsv.type_value, cdsv.zeal_id, cdsv.zeal_key, cdsv.zeal_name, cdsv.inductor_group, cdsv.zzzt_seggregation, (SELECT COUNT(*) FROM tttfapp.zealpremisis_mmte_data WHERE zealpremisis_data_id = dd.zealpremisis_data_id) AS associated_mmtes FROM ttsginfo.tttf_zeals_premisis_view cdsv INNER JOIN tttfapp.zeal_data dd ON dd.zeal_id = cdsv.zeal_id INNER JOIN tttfapp.zealpremisis_mmte_data dsrd ON dsrd.zealpremisis_data_id = dd.zealpremisis_data_id INNER JOIN tttfapp.scattered_mmte rr ON rr.scattered_mmte_id = dsrd.scattered_mmte_id AND rr.scattered_mmte_name = ? WHERE cdsv.zzzt_seggregation IS NULL UNION ALL SELECT DISTINCT cdsv.type_value, cdsv.zeal_id, cdsv.zeal_key, cdsv.zeal_name, cdsv.inductor_group, cdsv.zzzt_seggregation, (SELECT COUNT(*) FROM tttfapp.zealpremisis_mmte_data WHERE zealpremisis_data_id = ld.zealpremisis_data_id) AS associated_mmtes FROM ttsginfo.tttf_zeals_premisis_view cdsv INNER JOIN tttfapp.zzzt_data ld ON ld.zzzt_id = cdsv.zzzt_id INNER JOIN tttfapp.zealpremisis_mmte_data dsrd ON dsrd.zealpremisis_data_id = ld.zealpremisis_data_id INNER JOIN tttfapp.scattered_mmte rr ON rr.scattered_mmte_id = dsrd.scattered_mmte_id AND rr.scattered_mmte_name = ? WHERE cdsv.zzzt_seggregation IS NOT NULL) SELECT type_value AS "Type", zeal_id AS "Zeal ID", zeal_key AS "Zeal Key", zeal_name AS "Zeal Name", inductor_group AS "Inductor Group", zzzt_seggregation AS "Zzzt Seggregation", associated_mmtes AS "Scatteredd Mmtes" FROM combined_data ORDER BY CASE type_value WHEN 'ZEAL' THEN 1 WHEN 'PREMISIS' THEN 2 ELSE 3 END, zeal_key ASC, zzzt_seggregation NULLS RUST
