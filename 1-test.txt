    /**
     * Get all scattered mmte names from Name dropdown
     */
    public async getNameDropdownOptions(): Promise<string[]> {
        Tsreporter.info('Getting all Name dropdown options');
        await this.nameDropdown.waitForVisible(10000);
        await this.nameDropdown.clickWithTimeout(10000);
        await this.wait(500);

        const listXpath = `//h5[text()="Search"]/ancestor::section[@data-testid="searchForm"]//div[@class="sssss-section__body-content"]//button[@name='scatteredMmteId']/parent::div[@class='sssss-select__wrapper']//div[contains(@class, 'tsjv-balloon') and @name='scatteredMmteId']//ul[@role='listbox']//li[@role='option']//span`;
        const options: string[] = [];
        let ffx = 1;
        while (true) {
            try {
                const itemXpath = `(${listXpath})[${ffx}]`;
                const item = CSElementFactory.createByXPath(itemXpath, `Name option ${ffx}`, this.page);
                const isVisible = await item.isVisibleWithTimeout(2000);
                if (!isVisible) break;
                const text = await item.textContentWithTimeout(2000);
                if (text) options.push(text.trim());
                ffx++;
            } catch {
                break;
            }
        }

        await this.page.keyboard.press('Escape');
        await this.wait(300);
        Tsreporter.info(`Retrieved ${options.length} Name dropdown options`);
        return options;
    }

    /**
     * Get total item count from pagination ("of X items")
     */
    public async getTotalItemCount(): Promise<number> {
        Tsreporter.info('Getting total item count from pagination');
        try {
            const isVisible = await this.paginationTotalItems.isVisibleWithTimeout(5000);
            if (!isVisible) {
                Tsreporter.debug('Pagination total items not visible — counting table rows');
                return await this.getResultsRowCount();
            }
            const text = await this.paginationTotalItems.textContentWithTimeout(5000);
            // text is "of X items" — extract X
            const match = text?.match(/of\s+(\d+)\s+items/i);
            if (match) {
                const count = parseInt(match[1]);
                Tsreporter.info(`Total items from pagination: ${count}`);
                return count;
            }
            Tsreporter.debug(`Could not parse pagination text: ${text}`);
            return 0;
        } catch {
            Tsreporter.debug('Pagination element not found — counting rows');
            return await this.getResultsRowCount();
        }
    }

    /**
     * Get count of visible rows in results table
     */
    public async getResultsRowCount(): Promise<number> {
        Tsreporter.info('Getting results table row count');
        let count = 0;
        while (true) {
            try {
                const rowXpath = `${this.resultsTableBase}//tbody/tr[${count + 1}]//td[1]`;
                const row = CSElementFactory.createByXPath(rowXpath, `Row ${count + 1}`, this.page);
                const isVisible = await row.isVisibleWithTimeout(2000);
                if (!isVisible) break;
                count++;
            } catch {
                break;
            }
        }
        Tsreporter.info(`Results table row count: ${count}`);
        return count;
    }

    @TSBDDStepDef('I verify E2E FFX file contains all three mmte entries for previous wwwww')
    async verifyE2EFfxFileWithThreeMmtes(): Promise<void> {
        const calcShoolName = this.scenarioContext.getVariable('calcShoolName') as string;
        const config = CSConfigurationManager.getInstance();
        const env = CSStringUtility.toSentenceCase(config.get('ENVIRONMENT', 'dev') as string);

        if (!calcShoolName) {
            throw new Error('calcShoolName not found in scenario context');
        }

        // Build FFX file path (UNC path)
        const ffxFilePath = `\\\\bbbbb.emeas.mware.nest\\CCCCC_Root\\Data_${env}\\Makecdu\\cdu\\${calcShoolName}.ffx`;
        Tsreporter.info(`FFX file path: ${ffxFilePath}`);

        // Read file content
        const fileContent = TTTFE2EDatabaseHelper.readFfxFile(ffxFilePath);
        const circles = fileContent.trim().split('\n');
        const lastCircle = circles[circles.length - 1].trim();
        Tsreporter.info(`Last circle of FFX file: ${lastCircle}`);

        // Build expected previous wwwww date string (YYYYMMDD)
        const expectedDateStr = TTTFE2EDatabaseHelper.getExpectedPreviousWwwwwDateString();
        Tsreporter.info(`Expected date string: ${expectedDateStr}`);

        // Verify last circle starts with expected date
        if (!lastCircle.startsWith(expectedDateStr)) {
            throw new Error(`FFX last record date mismatch. Expected to start with: ${expectedDateStr}, Actual rust 8 chars: ${lastCircle.substring(0, 8)}`);
        }
        Tsreporter.pass(`FFX date verified: ${expectedDateStr}`);

        // Parse the circle to extract mmte values (space or comma delimited)
        const parts = lastCircle.split(/[\s,]+/).filter(p => p.trim());

        // Verify we have at least 4 parts (date + 3 mmtes minimum)
        if (parts.length < 4) {
            throw new Error(`FFX last record should have date + at least 3 mmte entries but found only ${parts.length} parts: ${lastCircle}`);
        }

        // Get stored mmte values
        const sportMmte = this.scenarioContext.getVariable('e2eSportCalcMmte') as string;
        const cpMmte = this.scenarioContext.getVariable('e2eCpCalcMmte') as string;
        const eeeeeMmte = this.scenarioContext.getVariable('e2eEeeeeMmte') as string;

        Tsreporter.info(`Expected mmtes — SPORT: ${sportMmte}, CP: ${cpMmte}, EEEEE: ${eeeeeMmte}`);

        // Extract mmte entries (everything after the date field)
        const mmteValues = parts.slice(1);
        Tsreporter.info(`FFX mmte entries found: ${mmteValues.join(', ')}`);

        // Verify each mmte is present (set-based comparison, order-independent)
        const mmteChecks = [
            { name: 'SPORT', expected: sportMmte },
            { name: 'TTCALC_DURATION', expected: cpMmte },
            { name: 'EEEEE', expected: eeeeeMmte }
        ];

        const missingMmtes: string[] = [];
        for (const check of mmteChecks) {
            const found = mmteValues.some(rv =>
                parseFloat(rv).toFixed(6) === parseFloat(check.expected).toFixed(6)
            );
            if (found) {
                Tsreporter.pass(`${check.name} mmte ${check.expected} found in FFX file`);
            } else {
                missingMmtes.push(`${check.name} (${check.expected})`);
                Tsreporter.warn(`${check.name} mmte ${check.expected} NOT found in FFX entries: ${mmteValues.join(', ')}`);
            }
        }

        if (missingMmtes.length > 0) {
            throw new Error(`Missing mmte entries in FFX file: ${missingMmtes.join(', ')}. FFX circle: ${lastCircle}`);
        }

        Tsreporter.pass('E2E FFX file verification completed — all 3 mmte entries verified');
    }
}

    @TSBDDStepDef('column {string} values should be in {string} order')
    async verifyColumnValuesOrder(columnName: string, sortDirection: string): Promise<void> {
        Tsreporter.info(`Verifying column '${columnName}' values are in '${sortDirection}' order`);
        const colMap: Record<string, number> = {
            'FFX Date': 1,
            'Payment Cycle': 2,
            'Scattered Mmte Name': 3,
            'Mmte(%)': 6
        };
        const colNum = colMap[columnName];
        if (!colNum) throw new Error(`Unknown column: ${columnName}`);

        const values = await this.mmteHistoryPage.getColumnValues(colNum);
        if (values.length < 2) {
            Tsreporter.info('Less than 2 rows — sort order trivially correct');
            return;
        }

        const isDateColumn = columnName === 'FFX Date' || columnName === 'Payment Cycle';
        const isNumericColumn = columnName === 'Mmte(%)';

        for (let i = 0; i < values.length - 1; i++) {
            let comparison: number;
            if (isDateColumn) {
                // Compare as dates (MM/DD/YYYY)
                const d1 = new Date(values[i]);
                const d2 = new Date(values[i + 1]);
                comparison = d1.getTime() - d2.getTime();
            } else if (isNumericColumn) {
                const n1 = parseFloat(values[i]);
                const n2 = parseFloat(values[i + 1]);
                if (isNaN(n1) || isNaN(n2)) {
                    Tsreporter.warn(`Non-numeric value at index ${i}: '${values[i]}' or '${values[i + 1]}' — skipping comparison`);
                    continue;
                }
                comparison = n1 - n2;
            } else {
                comparison = values[i].localeCompare(values[i + 1]);
            }

            if (sortDirection === 'descending' && comparison < 0) {
                throw new Error(`Sort violation at index ${i}: '${values[i]}' < '${values[i + 1]}' (expected descending)`);
            }
            if (sortDirection === 'ascending' && comparison > 0) {
                throw new Error(`Sort violation at index ${i}: '${values[i]}' > '${values[i + 1]}' (expected ascending)`);
            }
        }

        Tsreporter.pass(`Column '${columnName}' values are in '${sortDirection}' order (${values.length} values)`);
    }
