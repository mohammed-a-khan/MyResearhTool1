public static convertRoundingFactorToRawValue(factorValue: string): string {
        if (!factorValue || factorValue === '') return '';

        // If already a decimal number, remove leading zero if present (UI displays .1 not 0.1)
        if (/^[\d.]+$/.test(factorValue.trim())) {
            const trimmed = factorValue.trim();
            // Convert "0.1" to ".1", "0.01" to ".01", etc.
            if (trimmed.startsWith('0.')) {
                return trimmed.substring(1);
            }
            return trimmed;
        }

        // Mapping from fractional format to raw decimal value (UI displays without leading zero)
        const factorMapping: { [key: string]: string } = {
            '1/10th': '.1',
            'Nearest 1/10th': '.1',
            '1/100th': '.01',
            'Nearest 1/100th': '.01',
            '1/1000th': '.001',
            'Nearest 1/1000th': '.001',
            '1/10000th': '.0001',
            'Nearest 1/10000th': '.0001',
            '1/8th': '.125',
            'Nearest 1/8th': '.125',
            '1/16th': '.0625',
            'Nearest 1/16th': '.0625',
            '1/32nd': '.03125',
            'Nearest 1/32nd': '.03125'
        };

        return factorMapping[factorValue.trim()] || factorValue.trim();
    }
