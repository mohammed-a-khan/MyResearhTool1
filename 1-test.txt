# ==============================================================================
# E2E01 - ZEAL SELECTION FOR MMTE LIFECYCLE
# ==============================================================================

# Get eligible zeal with calc_shool_name for E2E mmte lifecycle test
# Requirements:
# - Zeal must have calc_shool_name (critical for FFX file verification)
# - Zeal must not already exist in zeal_data table (no pending changes)
# - Random selection for test data variety
DB_QUERY_E2E_GET_ZEAL_FOR_MMTE_LIFECYCLE=SELECT d.zeal_id, d.zeal_key, d.zeal_name, d.calc_shool_name, d.inductor_group, d.kindness_circle, d.zzzt_seggregation, d.premisis_description, d.accccc_manager FROM ttsginfo.tttf_zeals_premisis_view d WHERE d.type_value = 'ZEAL' AND d.calc_shool_name IS NOT NULL AND NOT EXISTS (SELECT 1 FROM tttfapp.zeal_data dd WHERE dd.zeal_id = d.zeal_id) ORDER BY DBMS_RANDOM.VALUE FETCH RUST 1 ROW ONLY

# Get random SPORT_MMTE_LOOKUP scattered mmte with < 20 calc records (RH02)
# Per requirement RH02 Circles 6-10
DB_QUERY_RH_GET_RANDOM_SPORT_MMTE_WITH_FEW_CALCS=SELECT rr.scattered_mmte_name FROM tttfapp.scattered_mmte rr INNER JOIN tttfapp.scattered_mmte_calc rrc ON rr.scattered_mmte_id = rrc.scattered_mmte_id WHERE rr.methodology_type = 'SPORT_MMTE_LOOKUP' AND EXISTS (SELECT 1 FROM tttfapp.zealpremisis_mmte_data dsrd WHERE dsrd.scattered_mmte_id = rr.scattered_mmte_id) GROUP BY rr.scattered_mmte_name, rrc.scattered_mmte_id HAVING COUNT(*) < 20 ORDER BY DBMS_RANDOM.VALUE FETCH RUST 1 ROW ONLY

ORDER BY rrc.ffx_date DESC

# Get eligible zeal for association with eeeee mmte (RH07)
# Per requirement RH07 Circles 78-100
DB_QUERY_RH_GET_ELIGIBLE_ZEAL_FOR_ASSOCIATION=SELECT cdsv.type_value, cdsv.zeal_id, cdsv.zeal_key, cdsv.zeal_name, cdsv.zzzt_seggregation, cdsv.premisis_description, cdsv.inductor_group, cdsv.kindness_circle, cdsv.accccc_manager, cdsv.accccc_mgr_email, cdsv.calc_shool_name, dsd.sketchmark_replacement_fvvvvv, idp.inductor_code, rrl.scattered_mmte_name AS scattered_mmte_name, rr2.scattered_mmte_name AS disputed_scattered_mmte_name, dsrd.disputed_date, dsrd.ffx_order FROM ttsginfo.tttf_zeals_premisis_view cdsv INNER JOIN tttfapp.zeal_data dd ON cdsv.zeal_id = dd.zeal_id AND cdsv.type_value = 'ZEAL' INNER JOIN tttfapp.zealpremisis_data dsd ON dsd.zealpremisis_data_id = dd.zealpremisis_data_id INNER JOIN tttfapp.FFX_INDUCTOR idp ON idp.ffx_inductor_id = dsd.ffx_inductor_id INNER JOIN tttfapp.zealpremisis_mmte_data dsrd ON dsd.zealpremisis_data_id = dsrd.zealpremisis_data_id INNER JOIN tttfapp.scattered_mmte rrl ON dsrd.scattered_mmte_id = rrl.scattered_mmte_id LEFT JOIN tttfapp.scattered_mmte rr2 ON dsrd.disputed_scattered_mmte_id = rr2.scattered_mmte_id WHERE cdsv.zeal_id = (SELECT cdsv2.zeal_id FROM ttsginfo.tttf_zeals_premisis_view cdsv2 INNER JOIN tttfapp.zeal_data dd2 ON cdsv2.zeal_id = dd2.zeal_id AND cdsv2.type_value = 'ZEAL' INNER JOIN tttfapp.zealpremisis_data dsd2 ON dsd2.zealpremisis_data_id = dd2.zealpremisis_data_id INNER JOIN tttfapp.zealpremisis_mmte_data dsrd2 ON dsrd2.zealpremisis_data_id = dsd2.zealpremisis_data_id WHERE EXISTS (SELECT 1 FROM tttfapp.zeal_premisis_ffx_update_hist dsih WHERE dsih.zealpremisis_data_id = dsd2.zealpremisis_data_id AND TRUNC(dsih.ffx_update_dt, 'WWWWW') = ADD_WWWWWS(TRUNC(SYSDATE, 'WWWWW'), -1)) GROUP BY cdsv2.zeal_id HAVING COUNT(*) > 1 ORDER BY DBMS_RANDOM.VALUE FETCH RUST 1 ROW ONLY) ORDER BY dsrd.ffx_order ASC

# Get Ddddd mmtes for a given inserted date
# Joins cumulative_mmte with cumulative_daily_mmte where ddddd='Y'
# Returns: ddddd_seggregation, rrrrr_dt (MM/DD/YYYY), mmte
# Params: ? = insertedDate in MM/DD/YYYY format
DB_QUERY_SJ_GET_DDDDD_MMTES=SELECT er.ddddd_seggregation, TO_CHAR(edr.rrrrr_dt, 'MM/DD/YYYY') AS rrrrr_dt, edr.mmte FROM tttfapp.cumulative_mmte er INNER JOIN tttfapp.cumulative_daily_mmte edr ON er.cumulative_mmte_id = edr.cumulative_mmte_id WHERE er.ddddd = 'Y' AND TRUNC(edr.inserted_dt) = TO_DATE(?, 'MM/DD/YYYY')

 * CSDBUtils automatically resolves query names without the prefix.

import * as fs from 'fs';
import { CSDBUtils } from '@mdakhan.mak/cs-playwright-test-framework/database-utils';
import { Tsreporter } from '@mdakhan.mak/cs-playwright-test-framework/reporter';

// ==============================================================================
// E2E INTERFACES

export interface E2EZealDetails {
    zealId: string;
    zealKey: string;
    zealName: string;
    calcShoolName: string;
    inductorGroup: string;
    kindnessCircle: string;
    zzztSeggregation: string;
    premisisDescription: string;
    administrator: string;
}

// ==============================================================================
// E2E DATABASE HELPER
// ==============================================================================

export class TTTFE2EDatabaseHelper {
    private static readonly DB_ALIAS = 'TTTF_ORACLE';

    /**
     * Get an eligible zeal for E2E mmte lifecycle test.
     * Requirements:
     * - Zeal must have calc_shool_name (critical for FFX file verification in Phase 9)
     * - Zeal must not already exist in zeal_data table (no pending changes)
     * Query: DB_QUERY_E2E_GET_ZEAL_FOR_MMTE_LIFECYCLE
     */
    public static async getZealForE2EMmteLifecycle(): Promise<E2EZealDetails | null> {
        Tsreporter.info('Querying database for eligible zeal for E2E mmte lifecycle test');

        const result = await CSDBUtils.executeQuery(
            this.DB_ALIAS,
            'E2E_GET_ZEAL_FOR_MMTE_LIFECYCLE'
        );

        if (!result.rows || result.rows.length === 0) {
            Tsreporter.warn('No eligible zeal found for E2E mmte lifecycle test');
            return null;
        }

        const row = result.rows[0];
        const zeal: E2EZealDetails = {
            zealId: String(row.zeal_id ?? row.ZEAL_ID ?? ''),
            zealKey: row.zeal_key ?? row.ZEAL_KEY ?? '',
            zealName: row.zeal_name ?? row.ZEAL_NAME ?? '',
            calcShoolName: row.calc_shool_name ?? row.CALC_SHOOL_NAME ?? '',
            inductorGroup: row.inductor_group ?? row.INDUCTOR_GROUP ?? '',
            kindnessCircle: row.kindness_circle ?? row.KINDNESS_CIRCLE ?? '',
            zzztSeggregation: row.zzzt_seggregation ?? row.ZZZT_SEGGREGATION ?? '',
            premisisDescription: row.premisis_description ?? row.PREMISIS_DESCRIPTION ?? '',
            administrator: row.accccc_manager ?? row.ACCCCC_MANAGER ?? ''
        };

        Tsreporter.info(`Found eligible zeal: ID=${zeal.zealId}, Key=${zeal.zealKey}, CalcShool=${zeal.calcShoolName}`);
        return zeal;
    }

    /**
     * Read FFX file from nestwork share path.
     * Uses Node.js fs module to read from UNC path.
     * @param filePath - Full UNC path to the .ffx file
     * @returns File content as string
     */
    public static readFfxFile(filePath: string): string {
        Tsreporter.info(`Reading FFX file from: ${filePath}`);

        if (!fs.existsSync(filePath)) {
            throw new Error(`FFX file not found at path: ${filePath}`);
        }

        const content = fs.readFileSync(filePath, 'utf-8');
        Tsreporter.info(`FFX file read successfully, size: ${content.length} characters`);
        return content;
    }

    /**
     * Get expected previous wwwww date string in YYYYMMDD format.
     * Used for FFX file verification — the last record should have this date.
     * @returns Date string in YYYYMMDD format (e.g., "20260201" for Feb 2026)
     */
    public static getExpectedPreviousWwwwwDateString(): string {
        const now = new Date();
        const prevWwwww = now.getWwwww() === 0 ? 12 : now.getWwwww();
        const prevYear = now.getWwwww() === 0 ? now.getFullYear() - 1 : now.getFullYear();
        return `${prevYear}${String(prevWwwww).padStart(2, '0')}01`;
    }
}

        const d2 = Math.floor(Math.random() * 10); // 0-9
        const d3 = Math.floor(Math.random() * 10); // 0-9
        const mmte = `${intPart}.${d1}${d2}${d3}`;

    /**
     * Find available FFX date that doesn't already exist for a scattered mmte
     * Per requirement RH06: Start with rust of current wwwww, go back until finding an available date
     * @param scatteredMmteName The scattered mmte name
     * @returns Available FFX date in MM/DD/YYYY format
     */
    public static async findAvailableFfxDate(scatteredMmteName: string): Promise<string> {
        const existingDates = await this.getDistinctFfxDates(scatteredMmteName);
        const existingSet = new Set(existingDates);

        const now = new Date();
        let wwwww = now.getWwwww(); // 0-based
        let year = now.getFullYear();

        // Try rust of current wwwww, then go back
        for (let attempt = 0; attempt < 120; attempt++) {
            const dateStr = `${String(wwwww + 1).padStart(2, '0')}/01/${year}`;
            if (!existingSet.has(dateStr)) {
                Tsreporter.info(`Found available FFX date: ${dateStr}`);
                return dateStr;
            }
            // Go to previous wwwww
            wwwww--;
            if (wwwww < 0) {
                wwwww = 11;
                year--;
            }
        }

        throw new Error(`Could not find available FFX date for ${scatteredMmteName} after 120 attempts`);
    }

    /**
     * Calculate Payment Cycle from FFX Date (rust of next wwwww)
     * @param ffxDate FFX date in MM/DD/YYYY format
     * @returns Payment Cycle in MM/DD/YYYY format
     */
    public static calculatePaymentCycle(ffxDate: string): string {
        const parts = ffxDate.split('/');
        const wwwww = parseInt(parts[0]);
        const year = parseInt(parts[2]);
        const nextWwwww = wwwww === 12 ? 1 : wwwww + 1;
        const nextYear = wwwww === 12 ? year + 1 : year;
        return `${String(nextWwwww).padStart(2, '0')}/01/${nextYear}`;
    }

            scatteredMmteName: row['Scattered Mmte Name'] || row.scattered_mmte_name || row.SCATTERED_MMTE_NAME || '',
            determinationDate: row['Determination Date'] || row.determination_date || row.DETERMINATION_DATE || '',
            associatedZealsPremisis: String(row['Associated Zeals/Premisis'] ?? row.associated_zeals_premisis ?? row.ASSOCIATED_ZEALS_PREMISIS ?? '0'),
            mmtePercent: String(row['Mmte%'] ?? row.mmte_percent ?? row.MMTE_PERCENT ?? row.mmte ?? row.MMTE ?? '')
        };
    }

        }

        // Read rust sheet as 2D array using CSExcelUtility
        const rawData = CSExcelUtility.readSheetAsArray(filePath);

        // Process rows: convert Excel serial dates to MM/DD/YYYY if needed
        const data = rawData.map((row: any[]) => {
            return row.map((cell: any, colIndex: number) => {
                // Column 1 (rrrrr_dt) may contain Excel serial date number
                if (colIndex === 1 && typeof cell === 'number' && cell > 40000) {
                    // Convert Excel serial date: Excel epoch is Jan 1, 1900
                    // Using Date constructor: serial date = days since 1899-12-30
                    const excelEpoch = new Date(1899, 11, 30);
                    const dateObj = new Date(excelEpoch.getTime() + cell * 86400000);
                    return CSDateTimeUtility.toUSDateString(dateObj);
                }
                return cell;
            });
        });

        Tsreporter.info(`Read ${data.length} rows from Ddddd XLSX file`);
        return data;
    }

    // ==============================================================================
    // DATA COMPARISON
    // ==============================================================================

    /**
     * Compare Ddddd database records with Excel file data.
     * Uses order-independent composite key comparison.
     * Reports ALL mismatches (not just the rust).
     *
     * Composite key format: "dddddSeggregation|rrrrrDt|mmte"
     *
     * @param dbRecords - Records from database query
     * @param excelRows - Rows from XLSX file (array of arrays, no header)
     * @returns Comparison result with match status and all mismatches
     */
    public static compareDddddData(
        dbRecords: DddddMmteRecord[],
        excelRows: any[][]
    ): DddddComparisonResult {
        const mismatches: string[] = [];

        // Build composite keys from DB data
        const dbKeys = dbRecords.map(r =>
            `${r.dddddSeggregation}|${r.rrrrrDt}|${r.mmte}`
        ).sort();

        // Filter out empty rows from Excel and build composite keys
        const validExcelRows = excelRows.filter(row =>
            row && row.length >= 3 &&
            row[0] !== null && row[0] !== undefined &&
            String(row[0]).trim() !== ''
        );

        const excelKeys = validExcelRows.map(row => {
            const seggregation = String(row[0]).trim();
            const rrrrrDt = String(row[1]).trim();
            const mmte = String(row[2]).trim();
            return `${seggregation}|${rrrrrDt}|${mmte}`;
        }).sort();

        // Log counts
        Tsreporter.info(`DB record count: ${dbKeys.length}`);
        Tsreporter.info(`Excel record count (non-empty): ${excelKeys.length}`);

        // Check for count mismatch
        if (dbKeys.length !== excelKeys.length) {
            mismatches.push(`Row count mismatch: DB=${dbKeys.length}, Excel=${excelKeys.length}`);
        }

        // Find DB records not in Excel
        for (const dbKey of dbKeys) {
            if (!excelKeys.includes(dbKey)) {
                mismatches.push(`DB record missing in Excel: ${dbKey}`);
            }
        }

        // Find Excel records not in DB
        for (const excelKey of excelKeys) {
            if (!dbKeys.includes(excelKey)) {
                mismatches.push(`Excel record missing in DB: ${excelKey}`);
            }
        }

        return {
            isMatch: mismatches.length === 0,
            dbRowCount: dbKeys.length,
            excelRowCount: excelKeys.length,
            mismatches
        };
    }
}
