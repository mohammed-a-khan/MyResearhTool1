/**
     * Wait for actual changes data to load (not "No data available" placeholder)
     * Uses generic waitForTableData from CSBasePage
     * @param timeout - Maximum wait time in milliseconds (default: 15000)
     * @returns true if data loaded, false if still showing "No data available"
     */
    public async waitForChangesDataToLoad(timeout: number = 15000): Promise<boolean> {
        const result = await this.waitForTableData(this.changesTable, 'No data available', timeout);
        return result.success;
    }
